#+TITLE: SQL Cheat Sheet

* Distinct
#+BEGIN_SRC sql
SELECT DISTINCT Country FROM Customers;
#+END_SRC


#+BEGIN_SRC sql
SELECT COUNT(DISTINCT Country)
FROM Customers;
#+END_SRC

#+BEGIN_SRC sql
SELECT COUNT(*) AS DistinctCountries
FROM (
        SELECT DISTINCT Country
        FROM Customers
     );
#+END_SRC 

* WHERE
#+BEGIN_SRC sql
SELECT <column1>, <column2>, ..., <columnN>
FROM   <table>
WHERE  <condition>
#+END_SRC
